<program> ::= <function>
<function> ::= "INT" "IDENTIFIER" "LEFT_PAREN" "RIGHT_PAREN" "LEFT_BRACE" <statement>* "RIGHT_BRACE"
<statement> ::= <returnStatement> | <exprStatement> | <declarationStatement>
<returnStatement> ::= "RETURN" <expr> "SEMICOLON"
<exprStatement> ::= <expr> "SEMICOLON"
<declarationStatement> ::= "INT" "IDENTIFIER" ("=" <expr>)? ";"
<expr> ::= <assignment>
<assignment> ::= "IDENTIFIER" "=" <assignment> | <addition>
<addition> ::= <unary> ( "+" <unary>)*
<unary> ::= "-" <unary> | <primary>
<primary> ::= "INTEGER" | "IDENTIFIER" | "(" <expr> ")"
